class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:

        # bigger word
        if len(word1) > len(word2):
            big_word = word1
        else:
            big_word = word2

        # create empty string for merged 
        merged_string = []

        # iterate over the larger word, use the index to append both words

        for l in range(0,len(big_word)):
            print(l)
            if l + 1 <= len(word1):  
                merged_string.append(word1[l])   
            if l + 1 <= len(word2):  
                merged_string.append(word2[l])
            
        merged_string = ''.join(merged_string)

        return merged_string

class Solution:
    def mergeAlternately(self, word1: str, word2: str) -> str:

        str_merged = ""

        if len(word1) >= len(word2):
            longest_word = word1
        else:
            longest_word = word2

        for i in range(len(longest_word)):
            if i+1 <= len(word1):
                str_merged += word1[i]
            
            if i+1 <= len(word2):
                str_merged += word2[i]
        
        return str_merged
