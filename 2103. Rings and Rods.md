# 2103. Rings and Rods
https://leetcode.com/problems/rings-and-rods/description/

```python
class Solution:
    def countPoints(self, rings: str) -> int:
        
        R = []
        G = []
        B = []

        for i in range(0, len(rings) -1,2):
            if rings[i] == 'R':
                R.append(rings[i+1])
            elif rings[i] == 'G':
                G.append(rings[i+1])
            elif rings[i] == 'B':
                B.append(rings[i+1])

        result = set(R).intersection(set(G), set(B))

        return len(result)
```
